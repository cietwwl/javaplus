package PACKAGE_NAME.utils;


import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Set;


public class Mapping {

	private static Map<String, String> table2class = new HashMap<String, String>();
	
	private static Map<String, String> class2table = new HashMap<String, String>();
	
	private static Map<String, Set<String>> columns = new HashMap<String, Set<String>>();
	
	static {
		SETS_CLASSS_TABLES
		SETS_TABLE_COLUMNS
	}
	
	public static String toTableName(String className) {
		return class2table.get(className);
	}	
	
	public static String toTableName(Class<?> clazz) {
		return class2table.get(clazz.getSimpleName());
	}
	
	public static String toClassName(String tableName) {
		return table2class.get(tableName);
	}
	
	public static Class<?> getDAOImplClass(String tableName) {
		try {
			return Class.forName("PACKAGE_NAME.dao.Impl." + Mapping.toClassName(tableName) + "DAOImpl");
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
			return null;
		}
	}
	
	public static Class<?> getClass(String tableName) {
		try {
			return Class.forName("PACKAGE_NAME.domain." + Mapping.toClassName(tableName));
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
			return null;
		}
	}

	public static Set<String> getAllTableName() {
		return table2class.keySet();
	}
	
	public static Set<String> getColumns(String tableName) {
		return columns.get(tableName);
	}
}
